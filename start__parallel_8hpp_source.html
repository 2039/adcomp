53 /*
54  Suppose we have a mapping F:R^n->R^m which may be written as F=F1+...+Fk.
55  Suppose we have tapes Fi:R^n->R^mi representing Fi with identical domain
56  but with *reduced range dimension* (because some range components of Fi
57  does not depend on any of the domain variables).
58  Based on these tape chunks construct an object behaving just like the
59  corresponding full taped version of F.
60  */
61 template <class Type>
62 struct parallelADFun:ADFun<Type>{ /* Inheritance just so that compiler wont complain about missing members */
63  typedef ADFun<Type> Base;
64  /* Following five members must be defined by constructor.
65  Outer vectors are indexed by the chunk number.
66  E.g. for tape number i vecind[i] is a vector of numbers in the
67  interval [0,1,...,range-1] telling how to embed this tapes range
68  in the full range.
69  */
70  int ntapes;
71  vector<Base*> vecpf;
72  vector<vector<size_t> > vecind;
73  size_t domain;
74  size_t range;
75  /* Following members are optional */
76  vector<sphess* > H_;
77  /* row and column indices */
78  vector<int> veci;
79  vector<int> vecj;
80  /* Constructor:
81  In the case of a vector of ADFun pointers we assume that
82  they all have equal domain and range dimensions.
83  */
84  parallelADFun(vector<Base*> vecpf_){
85  size_t n=vecpf_.size();
86  ntapes=n;
87  vecpf.resize(n);
88  for(size_t i=0;i<n;i++)vecpf[i]=vecpf_[i];
89  domain=vecpf[0]->Domain();
90  range=vecpf[0]->Range();
91  vecind.resize(n);
92  for(size_t i=0;i<n;i++){
93  vecind(i).resize(range);
94  for(size_t j=0;j<range;j++){
95  vecind(i)[j]=j;
96  }
97  }
98  }
99  /* Constructor:
100  In the case of a vector of sphess pointers the range dimensions are allowed
101  to differ. Based on sparseness pattern represented by each tape we must
102  compute "vecind" (for each tape), formally by matching the individual
103  sparseness patterns in a numbering of the full sparseness pattern (the union).
104  */
105  parallelADFun(vector<sphess* > H){
106  H_=H;
107  domain=H[0]->pf->Domain();
108  int n=H.size();
109  ntapes=n;
110  vecpf.resize(n);
111  vecind.resize(n);
112  for(int i=0;i<n;i++)vecpf[i]=H[i]->pf;
113  size_t kmax=0;
114  for(int i=0;i<n;i++){
115  //vecind[i]=(vector<size_t>(H[i]->i))+(vector<size_t>(H[i]->j))*domain;
116  vecind[i]=((H[i]->i).cast<size_t>())+((H[i]->j).cast<size_t>())*domain;
117  kmax+=vecind[i].size();
118  }
119  veci.resize(kmax);vecj.resize(kmax);
120  vector<int> pos(n); /* keep track of positions in individual index vectors */
121  for(int i=0;i<n;i++){pos(i)=0;};
122  if(config.trace.parallel) Rcout << "Hessian number of non-zeros:\n";
123  for(int i=0;i<n;i++){
124  if(config.trace.parallel) Rcout << "nnz = " << vecind(i).size() << "\n";
125  };
126  vector<size_t> value(n); /* value corresponding to pos */
127  int k=0; /* Incremented for each unique value */
128  size_t m; /* Hold current minimum value */
129  size_t inf=-1; /* size_t is unsigned - so -1 should give the largest possible size_t... */
130  int rowk=-1,colk=-1; /* -Wall */
131  while(true){
132  for(int i=0;i<n;i++){if(pos(i)<vecind(i).size())value(i)=vecind(i)[pos(i)];else value(i)=inf;}
133  m=value(0);
134  for(int i=0;i<n;i++){if(value(i)<m)m=value(i);}
135  if(m==inf)break;
136  for(int i=0;i<n;i++){
137  if(value(i)==m){
138  vecind(i)[pos(i)]=k;
139  rowk=(H[i]->i)[pos(i)];
140  colk=(H[i]->j)[pos(i)];
141  pos(i)++;
142  }
143  }
144  veci[k]=rowk;
145  vecj[k]=colk;
146  k++;
147  }
148  range=k;
149  //veci.resize(k);vecj.resize(k);
150  veci.conservativeResize(k);vecj.conservativeResize(k);
151  };

165  template <typename VectorBase>
166  VectorBase subset(const VectorBase& x, size_t tapeid, int p=1){
167  VectorBase y;
168  y.resize(vecind(tapeid).size()*p);
169  for(int i=0;i<y.size()/p;i++)
170  for(int j=0;j<p;j++)
171  {y(i*p+j)=x(vecind(tapeid)[i]*p+j);}
172  return y;
173  }
