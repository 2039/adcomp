60  Triangle<BlockType> operator*(Triangle<BlockType> other){
61  BlockType A, B;
62  A = (this->A) * (other.A);
63  B = (this->A) * (other.B);

65  return Triangle(A, B);
66  }

153  /*
154  Pade approximation of matrix exponential.
155  Can be applied to any of the previously implemented matrix
156  classes.
157  */
158  template<class matrix_pade>
159  matrix_pade expm(matrix_pade A){
160  double log2NormInf = log( A.norm() );
161  log2NormInf /= log(double(2.0));
162  double e = std::floor(log2NormInf) + 1.0;
163  double s = e + 1.0;
164  s = (s < 0 ? 0 : s);
165  matrix_pade AA = A.scale( 1.0 / pow( 2.0, s ) );
166  matrix_pade X = AA;
167  double c = 0.5;
168  matrix_pade E = AA.scale( c).addIdentity();
169  matrix_pade D = AA.scale(-c).addIdentity();
170  int q = 8; /* R uses 8 - ADMB 6 */
171  int p = 1;
172  for(int k = 2; k<=q; ++k){
173  c *= double(q-k+1) / double(k*(2*q-k+1));
174  X = AA * X;
175  matrix_pade cX = X.scale(c);
176  E += cX;
177  if (p == 1) { D += cX; } else { D -= cX; }
178  p = (p == 1) ? 0 : 1;
179  }
180  matrix_pade invD = D.inverse();
181  E = invD * E;
182  for(int k = 1; k<=s; k++){
183  E = E * E;
184  }
185  return E;
186  }
