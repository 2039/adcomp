237  piecewice(const vector<Type> &x_, const vector<Type> &y,
238  bool leftcontinuous_=true):
239  x(x_),leftcontinuous(leftcontinuous_){
240  n=x.size();
241  if(n!=y.size()-1)error("piecewice: y must be one longer than x");
242  dy.resize(n);
243  y0=y(0);
244  for(int i=0;i<n;i++)dy(i)=y(i+1)-y(i);
245  }
246  inline Type operator()(const Type &t){
247  Type res=y0;
248  if(leftcontinuous){
249  for(int i=0;i<n;i++)res+=CppAD::CondExpGt(t,x(i),dy(i),Type(0));
250  } else {
251  for(int i=0;i<n;i++)res+=CppAD::CondExpGe(t,x(i),dy(i),Type(0));
252  }
253  return res;
254  }
