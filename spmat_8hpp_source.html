83  I_.setIdentity();
84  /* Sparse representations */
85  Eigen::SparseMatrix<Type> A=asSparseMatrix(A_);
86  Eigen::SparseMatrix<Type> I=asSparseMatrix(I_);
87  /* Kronecker */
88  Eigen::SparseMatrix<Type> AxA=kronecker(A,A);
89  Eigen::SparseMatrix<Type> IxI=kronecker(I,I);
90  matrix<Type> vecI=I_.vec().matrix();

92  /* Solve system C*vecV=vecI.
93  Note: No assumptions on symmetry or eigenvalue-range of C. Therefore
94  rewrite it as (C'*C)*vecV=(C*vecI). Since (C'*C) is symmetric, the
95  LDL'-factorization can be applied.
96  */
97  Eigen::SimplicialLDLT< Eigen::SparseMatrix<Type> > ldlt(C.transpose()*C);
98  matrix<Type> z = ldlt.solve(C.transpose()*vecI);

100  for(int i=0;i<z.size();i++)V(i)=z(i);
