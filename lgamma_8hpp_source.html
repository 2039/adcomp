28  0.99999999999980993227684700473478,

30  -1259.13921672240287047156078755283,
31  771.3234287776530788486528258894,
32  -176.61502916214059906584551354,
33  12.507343278686904814458936853,
34  -0.13857109526572011689554707,
35  9.984369578019570859563e-6,
36  1.50563273514931155834e-7
37  };

39  int k;
40  Type Ag;

44  for(k=1; k<=8; k++) { Ag += lanczos_7_c[k]/(x+k); }
