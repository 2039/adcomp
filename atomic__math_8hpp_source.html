66  void integrand_D_incpl_gamma_shape(double *x, int nx, void *ex){
67  double* parms=(double*)ex;
68  double shape = parms[0];
69  double n = parms[1];
70  double logc = parms[2];
71  for(int i=0; i<nx; i++){
72  x[i] = exp( -exp(x[i]) + shape * x[i] + logc ) * pow(x[i], n);
73  }
74  }
75  /* n'th order derivative of (scaled) incomplete gamma wrt. shape parameter */
76  double D_incpl_gamma_shape(double x, double shape, double n, double logc){
77  if(n<.5){
78  return exp(logc + Rf_lgammafn(shape)) * Rf_pgamma(x, shape, 1.0, 1, 0);
79  }
80  double epsabs=1e-10;
81  double epsrel=1e-10;
82  double result1=0;
83  double result2=0;
84  double abserr=10000;
85  int neval=10000;
86  int ier=0;
87  int limit=100;
88  int lenw = 4 * limit;
89  int last=0;
90  int* iwork = (int*)malloc(limit * sizeof(int));
91  double* work = (double*)malloc(lenw * sizeof(double));
92  double ex[3];
93  ex[0] = shape;
94  ex[1] = n;
95  ex[2] = logc; /* Scale integrand with exp(logc) */
96  double bound; /* For indefinite integration */
97  int inf=-1; /* corresponds to (-Inf, bound) */
98  bound = log(Rf_fmin2(x,shape));
99  /* integrate -Inf...min(log(x),log(shape)) */
100  Rdqagi(integrand_D_incpl_gamma_shape, ex, &bound, &inf,
101  &epsabs, &epsrel,
102  &result1, &abserr, &neval, &ier,
103  &limit, &lenw, &last, iwork, work);
104  if(ier!=0){
105 #ifndef _OPENMP
106  warning("incpl_gamma (indef) integrate unreliable: x=%f shape=%f n=%f ier=%i", x, shape, n, ier);
107 #endif
108  }
109  /* integrate min(log(x),log(shape))...log(x) */
110  if(x>shape){
111  ier = 0;
112  double a = bound;
113  double b = log(x);
114  Rdqags(integrand_D_incpl_gamma_shape, ex, &a, &b,
115  &epsabs, &epsrel,
116  &result2, &abserr, &neval, &ier,
117  &limit, &lenw, &last, iwork, work);
118  if(ier!=0){
119 #ifndef _OPENMP
120  warning("incpl_gamma (def) integrate unreliable: x=%f shape=%f n=%f ier=%i", x, shape, n, ier);
121 #endif
122  }
123  }
124  free(iwork);
125  free(work);
126  return result1 + result2;
127  }
