40  void integrand_D_incpl_gamma_shape(double *x, int nx, void *ex){
41  double* parms=(double*)ex;
42  double shape = parms[0];
43  double n = parms[1];
44  double logc = parms[2];
45  for(int i=0; i<nx; i++){
46  x[i] = exp( -exp(x[i]) + shape * x[i] + logc ) * pow(x[i], n);
47  }
48  }
49  /* n'th order derivative of (scaled) incomplete gamma wrt. shape parameter */
50  double D_incpl_gamma_shape(double x, double shape, double n, double logc){
51  if(n<.5){
52  return exp(logc + lgammafn(shape)) * pgamma(x, shape, 1.0, 1, 0);
53  }
54  double epsabs=1e-10;
55  double epsrel=1e-10;
56  double result1=0;
57  double result2=0;
58  double abserr=10000;
59  int neval=10000;
60  int ier=0;
61  int limit=100;
62  int lenw = 4 * limit;
63  int last=0;
64  int* iwork = (int*)malloc(limit * sizeof(int));
65  double* work = (double*)malloc(lenw * sizeof(double));
66  double ex[3];
67  ex[0] = shape;
68  ex[1] = n;
69  ex[2] = logc; /* Scale integrand with exp(logc) */
70  double bound; /* For indefinite integration */
71  int inf=-1; /* corresponds to (-Inf, bound) */
72  bound = log(fmin2(x,shape));
73  /* integrate -Inf...min(log(x),log(shape)) */
74  Rdqagi(integrand_D_incpl_gamma_shape, ex, &bound, &inf,
75  &epsabs, &epsrel,
76  &result1, &abserr, &neval, &ier,
77  &limit, &lenw, &last, iwork, work);
78  if(ier!=0){
79 #ifndef _OPENMP
80  warning("incpl_gamma (indef) integrate unreliable: x=%f shape=%f n=%f ier=%i", x, shape, n, ier);
81 #endif
82  }
83  /* integrate min(log(x),log(shape))...log(x) */
84  if(x>shape){
85  ier = 0;
86  double a = bound;
87  double b = log(x);
88  Rdqags(integrand_D_incpl_gamma_shape, ex, &a, &b,
89  &epsabs, &epsrel,
90  &result2, &abserr, &neval, &ier,
91  &limit, &lenw, &last, iwork, work);
92  if(ier!=0){
93 #ifndef _OPENMP
94  warning("incpl_gamma (def) integrate unreliable: x=%f shape=%f n=%f ier=%i", x, shape, n, ier);
95 #endif
96  }
97  }
98  free(iwork);
99  free(work);
100  return result1 + result2;
101  }
