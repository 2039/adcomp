452  /* ================== For parallel Hessian computation
453  Need three different parallel evaluation modes:
454  (1) *Parallel mode* where a parallel region is evaluated iff
455  current_parallel_region == selected_parallel_region
456  (2) *Serial mode* where all parallel region tests are evaluated
457  to TRUE so that "PARALLEL_REGION" tests are effectively removed.
458  A negative value of "current_parallel_region" or "selected_parallel_region"
459  is used to select this mode (the default).
460  (3) *Count region mode* where statements inside "PARALLEL_REGION{...}"
461  are *ignored* and "current_parallel_region" is increased by one each
462  time a parallel region is visited.
463  NOTE: The macro "PARALLEL_REGION" is supposed to be defined as
464  #define PARALLEL_REGION if(this->parallel_region())
465  where the function "parallel_region" does the book keeping.
466  */
467  bool parallel_ignore_statements;
468  int current_parallel_region; /* Identifier of a code-fragment of user template */
469  int selected_parallel_region; /* Consider _this_ code-fragment */
470  int max_parallel_regions; /* Max number of parallel region identifiers,
471  e.g. max_parallel_regions=omp_get_max_threads();
472  probably best in most cases. */
473  bool parallel_region(){ /* Is this the selected parallel region ? */
474  bool ans;
475  if(current_parallel_region<0 || selected_parallel_region<0)return true; /* Serial mode */
476  ans = (selected_parallel_region==current_parallel_region) && (!parallel_ignore_statements);
477  current_parallel_region++;
478  if(max_parallel_regions>0)current_parallel_region=current_parallel_region % max_parallel_regions;
479  return ans;

1240 {

1243  if(!isNewList(parameters))error("'parameters' must be a list");
1244  if(!isEnvironment(report))error("'report' must be an environment");

1248  F.set_parallel_region(parallel_region);
1249  int n = F.theta.size();
1250  vector<bool> keepcol(n); // Scatter for fast lookup
1251  for(int i=0; i<n; i++){
1252  keepcol[i]=true;
1253  }
1254  for(int i=0; i<LENGTH(skip); i++){
1255  keepcol[INTEGER(skip)[i]-1]=false; // skip is R-index !
1256  }
1257 #define KEEP_COL(col) (keepcol[col])
1258 #define KEEP_ROW(row,col) ( KEEP_COL(row) & (row>=col) )
1259 
1260  /* Tape 1: Function R^n -> R */
1261  Independent(F.theta);
1262  vector< AD<AD<AD<double> > > > y(1);
1263  y[0] = F.evalUserTemplate();
1264  ADFun<AD<AD<double> > > tape1(F.theta, y);
1265 
1266  /* Tape 2: Gradient R^n -> R^n (and optimize) */
1267  vector<AD<AD<double> > > xx(n);
1268  for(int i=0; i<n; i++) xx[i] = CppAD::Value(F.theta[i]);
1269  vector<AD<AD<double> > > yy(n);
1270  Independent(xx);
1271  yy = tape1.Jacobian(xx);
1272  ADFun<AD<double > > tape2(xx,yy);
1273  if (config.optimize.instantly) tape2.optimize();
1274 
1275  /* Tape 3: Hessian R^n -> R^m (optimize later) */
1276  tape2.my_init(keepcol);
1277  int colisize;
1278  int m=0; // Count number of non-zeros (m)
1279  for(int i=0; i<int(tape2.colpattern.size()); i++){
1280  colisize = tape2.colpattern[i].size();
1281  if(KEEP_COL(i)){
1282  for(int j=0; j<colisize; j++){
1283  m += KEEP_ROW( tape2.colpattern[i][j] , i);
1284  }
1285  }
1286  }
1287  // Allocate index vectors of non-zero pairs
1288  vector<int> rowindex(m);
1289  vector<int> colindex(m);
1290  // Prepare reverse sweep for Hessian columns
1291  vector<AD<double> > u(n);
1292  vector<AD<double> > v(n);
1293  for(int i = 0; i < n; i++) v[i] = 0.0;
1294  vector<AD<double> > xxx(n);
1295  for(int i=0; i<n; i++) xxx[i]=CppAD::Value(CppAD::Value(F.theta[i]));
1296  vector<AD<double> > yyy(m);
1297  CppAD::vector<int>* icol;
1298  // Do sweeps and fill in non-zero index pairs
1299  Independent(xxx);
1300  tape2.Forward(0, xxx);
1301  int k=0;
1302  for(int i = 0; i < n; i++){
1303  if (KEEP_COL(i)) {
1304  tape2.myReverse(1, v, i /*range comp*/, u /*domain*/);
1305  icol = &tape2.colpattern[i];
1306  for(int j=0; j<int(icol->size()); j++){
1307  if(KEEP_ROW( icol->operator[](j), i )){
1308  rowindex[k] = icol->operator[](j);
1309  colindex[k] = i;
1310  yyy[k] = u[icol->operator[](j)];
1311  k++;
1312  }
1313  }
1314  }
1315  }

1317  ptape3->Dependent(xxx,yyy);
