SEXP asSEXP(const vector<Type> &a) \
{ \
int size = a.size(); \
SEXP val; \
PROTECT(val = allocVector(REALSXP,size)); \
double *p = REAL(val); \
for (int i = 0; i < size; i++) p[i] = asDouble(a[i]); \
UNPROTECT(1); \
return val; \
}
