892 template <class distribution1, class distribution2>
893 class SEPARABLE_t{
894  TYPEDEFS(typename distribution1::scalartype);
895 private:
896  distribution1 f;
897  distribution2 g;
898 public:
899  SEPARABLE_t(){}
900  SEPARABLE_t(distribution1 f_, distribution2 g_){f=f_;g=g_;}
901  /*
902  Example: x.dim=[n1,n2,n3].
903  Apply f on outer dimension (n3) and rotate:
904  [n3,n1,n2]
905  Apply g on new outer dimension (n2) and rotate back:
906  [n1,n2,n3]
907  */
908  arraytype jacobian(arraytype x){
909  int n=f.ndim();
910  x=f.jacobian(x);
911  x=x.rotate(n);
912  x=g.jacobian(x);
913  x=x.rotate(-n);
914  return x;
915  }
916  /* Create zero vector corresponding to the last n dimensions of dimension-vector d */
917  arraytype zeroVector(vector<int> d, int n){
918  int m=1;
919  vector<int> revd=d.reverse();
920  vector<int> revnewdim(n);
921  for(int i=0;i<n;i++){m=m*revd[i];revnewdim[i]=revd[i];}
922  vectortype x(m);
923  x.setZero();
924  return arraytype(x,revnewdim.reverse());
925  }
926  scalartype operator()(arraytype x){
927  if(this->ndim() != x.dim.size())std::cout << "Wrong dimension in SEPARABLE_t\n";
928  /* Calculate quadform */
929  arraytype y(x.dim);
930  y=jacobian(x);
931  y=x*y; /* pointwise */
932  scalartype q=scalartype(.5)*(y.sum());
933  /* Add normalizing constant */
934  int n=f.ndim();
935  arraytype zf=zeroVector(x.dim,n);
936  q+=f(zf)*(scalartype(x.size())/scalartype(zf.size()));
937  x=x.rotate(n);
938  int m=g.ndim();
939  arraytype zg=zeroVector(x.dim,m);
940  q+=g(zg)*(scalartype(x.size())/scalartype(zg.size()));
941  q-=log(sqrt(2.0*M_PI))*(zf.size()*zg.size());
942  /* done */
943  return q;
944  }
945  int ndim(){return f.ndim()+g.ndim();}
946  VARIANCE_NOT_YET_IMPLEMENTED;
947 
948  /* For parallel accumulation:
949  ==========================
950  Copied operator() above and added extra argument "i" to divide the accumulation
951  in chunks. The evaluation of
952  operator()(x)
953  is equivalent to summing up
954  operator()(x,i)
955  with i running through the _outer_dimension_ of x.
956  */
957  scalartype operator()(arraytype x, int i){
958  if(this->ndim() != x.dim.size())std::cout << "Wrong dimension in SEPARABLE_t\n";
959  /* Calculate quadform */
960  arraytype y(x.dim);
961  y=jacobian(x);
962  y=x*y; /* pointwise */
963  scalartype q=scalartype(.5)*(y.col(i).sum());
964  /* Add normalizing constant */
965  if(i==0){
966  int n=f.ndim();
967  arraytype zf=zeroVector(x.dim,n);
968  q+=f(zf)*(scalartype(x.size())/scalartype(zf.size()));
969  x=x.rotate(n);
970  int m=g.ndim();
971  arraytype zg=zeroVector(x.dim,m);
972  q+=g(zg)*(scalartype(x.size())/scalartype(zg.size()));
973  q-=log(sqrt(2.0*M_PI))*(zf.size()*zg.size());
974  }
975  /* done */
976  return q;
977  }
978 
979 
980 };

1053 class PROJ_t{
1054  TYPEDEFS(typename distribution::scalartype);
1055 private:
1056  distribution f;
1057  bool initialized;
1058 public:
1059  vector<int> proj;
1060  vector<int> cproj; /* complementary proj _sorted_ */
1061  int n,nA,nB;
1062  matrixtype Q; /* Full precision */
1063  MVNORM_t<scalartype> dmvnorm; /* mean zero gaussian with covariance Q_BB */
1064  PROJ_t(){}
1065  PROJ_t(distribution f_, vector<int> proj_){
1066  f=f_;
1067  proj=proj_;
1068  initialized=false;
1069  }
1070  void initialize(int n_){
1071  if(!initialized){
1072  n=n_;
1073  nA=proj.size();
1074  nB=n-nA;
1075  cproj.resize(nB);
1076  vector<int> mark(n);
1077  mark.setZero();
1078  for(int i=0;i<nA;i++)mark[proj[i]]=1;
1079  int k=0;
1080  for(int i=0;i<n;i++)if(!mark[i])cproj[k++]=i;
1081  // Full precision
1082  //matrixtype I(n,n);
1083  //I.setIdentity();
1084  //vectortype v(I);
1085 
1086  k=0;
1087  vectortype v(n*n);
1088  for(int i=0;i<n;i++)
1089  for(int j=0;j<n;j++)
1090  v(k++)=scalartype(i==j);
1091 
1092  vector<int> dim(2);
1093  dim << n,n;
1094  arraytype a(v,dim);
1095  a=f.jacobian(a);
1096  Q.resize(n,n);
1097  for(int i=0;i<n*n;i++)Q(i)=a[i];
1098  // Get Q_BB
1099  matrixtype QBB(nB,nB);
1100  for(int i=0;i<nB;i++)
1101  for(int j=0;j<nB;j++)
1102  QBB(i,j)=Q(cproj[i],cproj[j]);
1103  dmvnorm=MVNORM_t<scalartype>(QBB);
1104  }
1105  initialized=true;
1106  }
1107  vectortype projB(vectortype x){
1108  vectortype y(nB);
1109  for(int i=0;i<nB;i++)y[i]=x[cproj[i]];
1110  return y;
1111  }
1112  vectortype setZeroB(vectortype x){
1113  for(int i=0;i<nB;i++)x[cproj[i]]=scalartype(0);
1114  return x;
1115  }
1116  scalartype operator()(vectortype x){
1117  initialize(x.size());
1118  x=setZeroB(x);
1119  vector<int> dim(1);
1120  dim << x.size();
1121  arraytype xa(x,dim);
1122  vectortype y=projB(f.jacobian(xa));
1123  // f(x_A,0) - dmvnorm(y_B) + 2*dmvnorm(0)
1124  return f(xa) - dmvnorm(y) + 2*dmvnorm(y*scalartype(0));
1125  }
1126  /* array versions */
1127  arraytype projB(arraytype x){
1128  vectortype z((x.size()/n)*nB);
1129  vector<int> dim(x.dim);
1130  dim[dim.size()-1]=nB;
1131  arraytype y(z,dim);
1132  for(int i=0;i<nB;i++)y.col(i)=x.col(cproj[i]);
1133  return y;
1134  }
1135  arraytype setZeroB(arraytype x){
1136  for(int i=0;i<nB;i++)x.col(cproj[i])=x.col(0)*scalartype(0);
1137  return x;
1138  }
1139  arraytype jacobian(arraytype x){
1140  initialize(x.dim[x.dim.size()-1]);
1141  arraytype xa=setZeroB(x);
1142  arraytype y=projB(f.jacobian(xa));
1143  // WRONG: ----> return f.jacobian(xa) - dmvnorm.jacobian(y);
1144  // y=P*Q*Z*x so should be (P*Q*Z)' * dmvnorm.jacobian(y).
1145  // Note: only P is not symmetric.
1146  arraytype tmp=f.jacobian(xa);
1147  arraytype tmp0=tmp*scalartype(0);
1148  arraytype tmp2=dmvnorm.jacobian(y);
1149  // apply P'
1150  for(int i=0;i<nB;i++){
1151  tmp0.col(cproj[i])=tmp2.col(i);
1152  }
1153  // apply Q'(=Q)
1154  tmp0=f.jacobian(tmp0);
1155  // apply Z'(=Z)
1156  tmp0=setZeroB(tmp0);
1157  // Done:
1158  return tmp-tmp0;
1159  }
1160  int ndim(){return f.ndim();}
1161  VARIANCE_NOT_YET_IMPLEMENTED;
1162 };
